dat.GUI使用教程

2019-03-14 18:11:35

本文用来介绍dat.GUI的使用方法。

# 介绍

# dat.gui是什么

    dat.GUI 是一个轻量级的图形用户界面库（GUI 组件），使用这个库可以很容易地创建出能够改变代码变量的界面组件。

# dat.gui能做什么

    主要用来对页面参数的调整，并且很容易上手。

献上官方教程（英文）

# 正文

# 引入

// 第一种方法，直接通过cdn引入
<script src="https://cdn.bootcss.com/dat-gui/0.7.5/dat.gui.min.js"></script>

// 第二种方法
// npm安装
npm install --save dat.gui
// 引入
const dat = require('dat.gui');

# 基本配置

<script type="text/javascript">

var FizzyText = function() {
  this.message = 'dat.gui';//String类型渲染为input输入框
  this.speed = 0.8;//Number类型渲染为slider
  this.displayOutline = false;//Boolean类型渲染为checkbox
  this.fun = function() {
        console.log("you click me");
  };//Function类型渲染为button
};

window.onload = function() {
  var fizzyText = new FizzyText();
  var gui = new dat.GUI();
  gui.add(fizzyText, 'message');
  gui.add(fizzyText, 'speed', -5, 5);
  gui.add(fizzyText, 'displayOutline');
  gui.add(fizzyText, 'fun');
};

</script>

var FizzyText = function () {
    this.noiseStrength = 5;
    this.growthSpeed = 3;
    this.maxSize = 6;
    this.message = "pizza";
    this.speed = 0;
};

gui.add(fizzyText, 'noiseStrength').step(5); //步长，即输入数据只能为5的整除数
gui.add(fizzyText, 'growthSpeed', -5, 5); // 范围限制，对应最小值，最大值
gui.add(fizzyText, 'maxSize').min(0).step(0.25); // 综合运用

// 下拉菜单（接收数组或对象）
gui.add(fizzyText, 'message', [ 'pizza', 'chrome', 'hooray' ] );
gui.add(fizzyText, 'speed', { Stopped: 0, Slow: 0.1, Fast: 5 } );

# 预设

var FizzyText = function () {
    this.number = 1;
};


// 可以通过JSON导入预设配置
var fizzyText = new FizzyText();
var gui = new dat.GUI({
    load: {
        "preset": "mine",
        "remembered": {
            "Default": {
                "0": {
                    "number": 1
                }
            },
            "mine": {
                "0": {
                    "number": 123
                }
            }
        }
    }

gui.remember(fizzyText);
gui.add(fizzyText, 'number').min(0).max(10).step(1);

# 事件

# 自定义放置

// 将gui DOM节点放入一个新创建的节点中，就可以更改控制面板的位置了
gui.add(fizzyText, 'number').min(0).max(10).step(1);

var customContainer = document.getElementById('my-gui-container');
customContainer.appendChild(gui.domElement);

# 更新

// 自动刷新
var FizzyText = function () {
    this.number = 1;
};

window.onload = function () {
    var fizzyText = new FizzyText();
    var gui = new dat.GUI();

    gui.add(fizzyText, 'number', 0, 100).listen();

    var update = function () {
      // 也可以设置setTimeout或者setInterval进行定时刷新
        requestAnimationFrame(update);
        fizzyText.number = Math.random();
    };
    update();
};


// 手动刷新
var FizzyText = function () {
    this.number = 1;
};

window.onload = function () {
    var fizzyText = new FizzyText();
    var gui = new dat.GUI();

    gui.add(fizzyText, 'number', 0, 100);

    var update = function () {

        requestAnimationFrame(update);
        fizzyText.number = Math.cos(new Date().getTime());

        for (var i in gui.__controllers) {
            gui.__controllers[i].updateDisplay();
        }

    };

    update();
};
